    **IMPLEMENTATION DETAILS:**

    The seed script structure should follow Pattern 4 from the research:

    ```typescript
    import { neon } from '@neondatabase/serverless';
    import { drizzle } from 'drizzle-orm/neon-http';
    import { config } from 'dotenv';
    import { sql } from 'drizzle-orm';
    import { weekends, workItems, doneCriteria } from './schema';

    config({ path: '.env.local' });

    const client = neon(process.env.DATABASE_URL!);
    const db = drizzle({ client });

    async function seed() {
      console.log('Seeding database...');

      // 1. Clear existing data (order matters: children first)
      await db.delete(doneCriteria);
      await db.delete(workItems);
      await db.delete(weekends);

      // 2. Insert each weekend with .returning() to get the ID
      // 3. Insert work items referencing the weekend ID
      // 4. Insert done criteria referencing the weekend ID

      // REPEAT FOR ALL 11 WEEKENDS

      // 5. Reset identity sequences (CRITICAL)
      await db.execute(sql`SELECT setval(pg_get_serial_sequence('weekends', 'id'), (SELECT COALESCE(MAX(id), 1) FROM weekends))`);
      await db.execute(sql`SELECT setval(pg_get_serial_sequence('work_items', 'id'), (SELECT COALESCE(MAX(id), 1) FROM work_items))`);
      await db.execute(sql`SELECT setval(pg_get_serial_sequence('done_criteria', 'id'), (SELECT COALESCE(MAX(id), 1) FROM done_criteria))`);
      await db.execute(sql`SELECT setval(pg_get_serial_sequence('scorecard_ratings', 'id'), (SELECT COALESCE(MAX(id), 1) FROM scorecard_ratings))`);

      console.log('Seeding completed!');
    }

    seed().catch((err) => {
      console.error('Seeding failed:', err);
      process.exit(1);
    });
    ```

    Also import and clear `scorecardRatings` even though we don't seed it — it has a foreign key to weekends and must be cleared before weekends can be deleted.

    Use `COALESCE(MAX(id), 1)` in setval to handle edge cases where a table might be empty.
